CREATE TABLE computer( 
`id_computer` INT NOT NULL AUTO_INCREMENT ,
`netname` VARCHAR(250) NOT NULL ,
`motherboard` varchar(11) NOT NULL , 
`RAM_capacity` int(11) NOT NULL ,
`HDD_capacity` int(11) NOT NULL,
`monitor` varchar(11) NOT NULL,
`vendor` varchar(20) NOT NULL,
`quarantii` varchar(250) NOT NULL,
`FID_processor` int NOT NULL,
CONSTRAINT processor_fk FOREIGN KEY (FID_Processor) REFERENCES processor(ID_processor),
PRIMARY KEY (`id`))

create table Ward(
`id_ward` INT NOT NULL AUTO_INCREMENT ,
`name` VARCHAR(11) NOT NULL ,
PRIMARY KEY (`id`))


create table Nurse_Ward(
FID_Nurse int,
FID_Ward int,
CONSTRAINT Nurse_fk FOREIGN KEY (FID_Nurse) REFERENCES nurse(ID_Nurse),
CONSTRAINT Ward_fk2 FOREIGN KEY (FID_Ward) REFERENCES Ward (ID_Ward))

create table computer_software(
FID_computer int,
FID_software int,
CONSTRAINT computer_fk FOREIGN KEY (FID_computer) REFERENCES computer(ID_computer),
CONSTRAINT software_fk2 FOREIGN KEY (FID_software) REFERENCES software(ID_software))
